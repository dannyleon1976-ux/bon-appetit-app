var SPREADSHEET_ID = "1JOSwLdeQIJC1tG2d2fST-mlXWn-vPo5C80zgJIf6QF0";

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    var fechaActual = new Date();
    var fechaFormateada = Utilities.formatDate(fechaActual, Session.getScriptTimeZone(), "dd/MM/yyyy");
    var horaFormateada = Utilities.formatDate(fechaActual, Session.getScriptTimeZone(), "hh:mm:ss a");
    var action = data.action || "";
    
    if (action === "save_order") {
      var sheet = ss.getSheetByName("Pedido");
      if (!sheet) { sheet = ss.insertSheet("Pedido"); }
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(["Fecha", "Hora", "Cliente", "Productos", "Cantidad", "Precio Unitario", "Total"]);
        sheet.getRange(1, 1, 1, 7).setFontWeight("bold").setBackground("#F3F3F3");
      }
      var cliente = data.cliente || "Cliente Sin Nombre";
      var items = data.items || [];
      for (var i = 0; i < items.length; i++) {
        var item = items[i];
        sheet.appendRow([
          fechaFormateada,
          horaFormateada,
          cliente,
          item.name || "",
          Number(item.quantity) || 1,
          Number(item.price) || 0,
          (Number(item.price) || 0) * (Number(item.quantity) || 1)
        ]);
      }
      return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "Pedido guardado."}))
        .setMimeType(ContentService.MimeType.JSON);
    } else if (action === "save_client" || action === "update_client") {
      var sheet = ss.getSheetByName("Clientes");
      if (!sheet) { sheet = ss.insertSheet("Clientes"); }
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(["ID", "CLIENTE", "RIF CEDULA", "DIRECCION P REFENCIA", "TELEFONOS", "CONTACTO", "VENDEDOR", "CONDICION", "SECTOR", "ETIQUETA", "ENLACE netlify", "ENLACE github"]);
        sheet.getRange(1, 1, 1, 12).setFontWeight("bold").setBackground("#D0E1FD");
      }
      
      var idBuscado = data.id || data.idClient || data.row || "";
      var clienteBuscado = (data.cliente || "").toString().toLowerCase().trim();
      var foundRow = -1;
      
      var dataValues = sheet.getDataRange().getValues();
      for (var i = 1; i < dataValues.length; i++) {
        // Buscar por ID (Columna A)
        if (idBuscado && dataValues[i][0] == idBuscado) {
          foundRow = i + 1;
          break;
        }
        // Fallback: Buscar por Nombre (Columna B) si el ID no coincide o no se envió
        if (dataValues[i][1] && dataValues[i][1].toString().toLowerCase().trim() === clienteBuscado) {
          foundRow = i + 1;
        }
      }
      
      var rowData = [
        idBuscado || Utilities.getUuid(),
        data.cliente || "",
        data.rif_cedula || "",
        data.direccion_p_referencia || "",
        data.telefonos || "",
        data.contacto || "",
        data.vendedor || "",
        data.condicion || "",
        data.sector || "",
        data.etiqueta || "",
        data.netlify_url || "",
        data.github_url || ""
      ];
      
      if (foundRow > -1) {
        sheet.getRange(foundRow, 1, 1, 12).setValues([rowData]);
        return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "Cliente actualizado."}))
          .setMimeType(ContentService.MimeType.JSON);
      } else {
        sheet.appendRow(rowData);
        return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "Cliente registrado."}))
          .setMimeType(ContentService.MimeType.JSON);
      }
    }
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    var action = e.parameter.action || "get_clients";
    var sheetName = (action === "get_products") ? "Productos" : "Clientes";
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet || sheet.getLastRow() <= 1) {
      return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);
    }
    var values = sheet.getDataRange().getValues();
    var headers = values[0];
    var result = [];
    for (var i = 1; i < values.length; i++) {
      var row = values[i];
      var obj = {};
      for (var j = 0; j < headers.length; j++) {
        var key = headers[j].toString().trim().toLowerCase().replace(/ /g,"_").replace(/á/g,"a").replace(/é/g,"e").replace(/í/g,"i").replace(/ó/g,"o").replace(/ú/g,"u");
        obj[key] = row[j];
      }
      result.push(obj);
    }
    return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
